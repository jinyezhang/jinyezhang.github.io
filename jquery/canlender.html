canlender 脚本的构建如下：

var Canlender = function(opts) {
  this.el = document.getElementById(opts.el);
  //获取时间
  this.oDate = new Date();
  //获取当前的年份
  this.iYear = opts.year ? opts.year : this.oDate.getFullYear();
  //获取当前的月份
  this.iMonth = opts.month ? opts.month : this.oDate.getMonth();
  //获取当前的日
  this.iDay = this.oDate.getDate();
  //获取当前月份的第一天的具体时间
  this.oMonthDate = new Date(this.iYear, this.iMonth, 1);
  //获取当前月份第一天是星期几
  this.iWeek = this.oMonthDate.getDay();
  //获取月份天数集合
  this.aMonthDay = new Array(
    31,
    28 + this.isLeep(this.iYear),
    31,
    30,
    31,
    30,
    31,
    31,
    30,
    31,
    31,
    30
  );
  //获取当前月的天数
  this.aMonthDays = this.aMonthDay[this.iMonth];
  this.init();
};
Canlender.prototype = {
  init: function() {
    var str = "",
      _this = this,
      j,
      m,
      n;
    str = `<div class="week-wrap">
        <div class="week-day">星期日</div>
        <div class="week-day">星期一</div>
        <div class="week-day">星期二</div>
        <div class="week-day">星期三</div>
        <div class="week-day">星期四</div>
        <div class="week-day">星期五</div>
        <div class="week-day">星期六</div>
    </div>`;
    str += `<div class="day-wrap">
        <div class="day-col">`;
    //补充前面的，月份第一天星期几之前的都是灰色的
    var preDays = this.aMonthDay[this.iMonth - 1];
    for (j = 0; j < this.iWeek; j++) {
      str += `<div class="day pre">${preDays - this.iWeek + j + 1}</div>`;
    }
    //补全中间的，都是月份的每天的渲染
    for (m = 0; m < this.aMonthDays; m++) {
      if ((this.iWeek + m) % 7 == 0) {
        str += `</div><div class="day-col">`;
      }
      if (m + 1 == this.iDay) {
        str += `<div class="day now">${m + 1}</div>`;
      } else {
        str += `<div class="day">${m + 1}</div>`;
      }
    }
    var nextDays = 7 - ((this.iWeek + this.aMonthDays) % 7);
    //补全后面的，35减去iWeek减去当前月份的天数
    for (n = 0; n < nextDays; n++) {
      str += `<div class="day pre">${n + 1}</div>`;
    }
    str += `</div>
    </div>`;
    _this.el.innerHTML = str;
  },
  isLeep(year) {
    //判断是否是闰年
    return year % 4 == 0 ? (year % 400 == 0 ? 1 : 0) : year % 4 == 0 ? 1 : 0;
  },
};
