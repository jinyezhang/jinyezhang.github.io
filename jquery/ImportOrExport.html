# 1.导入文件——以文档流的方式传递给后端

这种的形式，需要点击导入按钮，然后打开选择文件的弹窗，选择文件后，获取文件的文档流。

<div style="opacity: 0">
  <input
    v-if="fileVisible"
    name="file"
    ref="fileUpload"
    style="width: 0px; height: 0px"
    accept=".csv, application/vnd.ms-excel, application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"
    type="file"
    v-on:change="changeFile($event)"
  />
</div>

changeFile(file) {
      var files = file.target.files[0];
      if (
        files.type ==
          "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet" ||
        files.type == "application/vnd.ms-excel" ||
        files.type == ".csv"
      ) {
        console.log("格式正确");
      } else {
        this.$message.error("文件格式错误！");
        return;
      }
      var formdata = new FormData();
      formdata.append("StreamContent", new Blob([files]));
      //导入文件的接口——axios的形式
      importFile(this.detailObj.orderNo, this.fileType=0, formdata, {
        Headers: {
          "Content-Type": "multipart/form-data",
          withCredentials: true,
        },
      }).then((res) => {
        //导入文件成功的操作
      });
      //导入文件——fetch的形式
      var baseUrl = process.env.VUE_APP_API_BASE_URL_API;//获取baseUrl
      var myHeaders = new Headers();//头部要加入用户的信息等
      var Authorization = Cookie.get("Authorization");//因为之前存储到cookie中，因此直接从cookie中获取
      myHeaders.append("Authorization", Authorization);
      var formdata = new FormData();
      formdata.append("StreamContent", new Blob([files]));
      var requestOptions = {
        method: "post",
        headers: myHeaders,
        body: formdata,
        redirect: "follow",//跟随重定向的地址,继续请求
      };
      fetch(
        `${baseUrl}/api/sheet-metal/orders/${this.detailObj.orderNo}/import?BomType=${this.fileType}`,
        requestOptions
      )
        .then((response) => {
        //导入文件成功的操作
        })
        .catch((error) => console.log("error", error));
      file.target.value = "";//这句可以实现允许多次上传重复的文件
    },

# 2.后端以文档流的形式返回，需要下载到本地

swagger 上面的下载文档流是下面的方式：

以 fetch 的形式下载文档流如下：

var param = this.filterParams(params);
var baseUrl = process.env.VUE_APP_API_BASE_URL_API;
var myHeaders = new Headers();
var Authorization = Cookie.get("Authorization");
myHeaders.append("Authorization", Authorization);
var requestOptions = {
  method: "get",
  headers: myHeaders,
  redirect: "follow",
};
fetch(`${baseUrl}/api/sheet-metal/orders/export?${param}`, requestOptions)
  .then((response) => {
    //如果是文档流的形式，则需要通过response.blob()的方式来转化，其他的形式都是response.json()的方式来转化的
    return response.blob();
  })
  .then((res) => {
    //拿到文档流之后，通过a标签的形式进行下载
    const link = document.createElement("a");
    let blob = new Blob([res], {
      type:
        ".csv, application/vnd.openxmlformats-officedocument.spreadsheetml.sheet, application/vnd.ms-excel'",
    });
    link.href = URL.createObjectURL(blob);
    link.download = `order.xlsx`; //在浏览器状态栏展示的文档名称
    link.click();
  })
  .catch((error) => console.log("error", error));
