# 递归

我的 js 基础很薄弱，因此在遇到问题的时候，我很郁闷。不知道咋写……

下面的这个递归操作，是我花了 2 天时间才想明白的。。。

后端返回的数据格式跟我想要的不一样：

根据上图可以知道，后端返回我的数据全是并列的兄弟数据，而我最终要展示的效果如下：

如果要实现上图中的效果，则需要保证数据非并列，而是children的父子级展示。

因此我需要每 5 个兄弟元素截断，然后将后续分别作为子级元素添加到以上兄弟元素的 children 中，依次类推。

因此我先要在数据中拿出 5 个作为子级，然后将剩余的数据，分别往子级的 children 中赋值。具体要不要插入到 children 中，可以找到没有子级 children 的进行赋值。

//递归查找没有子级的元素
recurrenceData(arr, data) {
    var i = 0;
    arr && arr.forEach(a => {
        if (a.children&&a.children.length) {
        //递归查找最后一级没有children的子级，找到就要赋值一个数据
            this.recurrenceData(a.children,data);
        } else {
            if (data.length > 0) {
                a.children = data.splice(0, 1);
            }
        }
    })
},
handleData(dataArr, n) {
    dataArr && dataArr.forEach(child => {
        if (child.children && child.children.length > 5) {
            var data = child.children.splice(5, child.children.length - 1);
            child.children && child.children.map(c => {
                if (data.length) {
                    c.children = data.splice(0, 1);
                }
            })
            //如果一行超过5个，则需要转行，就是把剩余的数据放到前5项里面的children中
            var row = Math.ceil(data.length / 5);
            //具体递归函数需要执行多少次，跟行数有关，每行5个，就可以得出需要执行的行数了
            for (var i = 0; i < row; i++) {
                if (data.length) {
                    child.children && child.children.map(c => {
                    //这个就是递归函数
                        this.recurrenceData(c.children, data);
                    })
                }
            }
        }
    })
    return dataArr;
}
